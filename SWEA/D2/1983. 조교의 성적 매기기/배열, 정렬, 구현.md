T = int(input())


for test_case in range(1, T + 1):

    n, k = map(int, input().split())

    score = ['A+', 'A0', 'A-', 'B+', 'B0', 'B-', 'C+', 'C0', 'C-', 'D0']

    score_list = []

    m = n//10



    for _ in range(n):

        mid, final, hw = map(int, input().split())

        calc = mid*0.35 + final*0.45 + hw*0.2

        score_list.append(calc)



    kscore = score_list[k-1]

    score_list.sort(reverse=True)

    kidx = score_list.index(kscore)

    ans = score[score_idx]

    

    print(f"#{test_case} {ans}")
