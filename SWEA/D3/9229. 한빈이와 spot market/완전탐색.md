```python
T = int(input())

for test_case in (1, T+1):
    N, M = map(int, input().split())
    snacks = list(map(int, input().split()))
    snacks.sort(reverse=True)
    maxw = 0
    for i in range(N):
        for j in range(i+1,N): #핵심!
            partw = sum(snacks[i], snacks[j])
            if (partw <= M) and (partw > maxw):
                maxw = partw
    print(f"#{test_case} {maxw}")
```
